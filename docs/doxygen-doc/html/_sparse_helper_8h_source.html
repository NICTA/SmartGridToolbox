22  /// @brief Object to efficiently construct large sparse matrices in an incremental manner.
23  ///
24  /// Use insert(...) to add elements, and when ready, get() to construct and return the sparse matrix.
25  /// @ingroup Utilities
26  template<typename T> class SparseHelper
27  {
28  public:
29 
30  SparseHelper(arma::uword nRow, arma::uword nCol, bool sort = true, bool add = true,
31  bool checkZeros = false, unsigned int nBatch = (1 << 16)) :
32  locs_(2, nBatch, arma::fill::none),
33  vals_(nBatch, arma::fill::none),
34  sort_(sort),
35  add_(add),
36  checkZeros_(checkZeros),
37  m_(nRow, nCol)
38  {
39  // Empty.
40  }
41 
42  void insert(arma::uword iRow, arma::uword iCol, const T& x)
43  {
44  if (nWaiting_ == vals_.size())
45  {
46  flush();
47  }
48 
49  locs_(0, nWaiting_) = iRow;
50  locs_(1, nWaiting_) = iCol;
51  vals_(nWaiting_) = x;
52  ++nWaiting_;
53  }
54 
55  arma::SpMat<T> get()
56  {
57  flush();
58  return std::move(m_); // Should apply move constructor.
59  }
60 
61  private:
62 
63  void flush()
64  {
65  if (nWaiting_ > 0)
66  {
67  arma::span sp1(0, 1);
68  arma::span sp2(0, nWaiting_ - 1);
69  m_ += arma::SpMat<T>(add_, locs_(sp1, sp2), vals_(sp2), m_.n_rows, m_.n_cols, sort_, checkZeros_);
70  nWaiting_ = 0;
71  }
72  }
73 
74 
75  private:
76 
77  arma::uword nWaiting_{0};
78  arma::Mat<arma::uword> locs_;
79  arma::Col<T> vals_;
80  bool sort_;
81  bool add_;
82  bool checkZeros_;
83  arma::SpMat<T> m_;
84  };
85 }
