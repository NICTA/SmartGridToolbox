28  /// The `time_series_zip` keyword adds a TimeSeriesZip to the Simulation.
29  ///
30  /// ~~~{.yaml}
31  /// - time_series_zip:
32  /// id: <string> # Component ID.
33  /// bus_id: <string> # Attached to this bus.
34  /// sim_network_id: <string> # Part of this SimNetwork.
35  /// time_series_id: <string> # Use this TimeSeries for values.
36  /// dt: <hh:mm:ss> # Update frequency.
37  /// matrix_elements: <int_mat> # n x 2 matrix corresponding to matrix elems of target ZIP matrix.
38  /// # Any combination of the following three elements:
39  /// data_indices_Y: <int_vec> # Data elements that will template onto ZIP matrix for const Y.
40  /// data_indices_I: <int_vec> # Data elements that will template onto ZIP matrix for const I.
41  /// data_indices_S: <int_vec> # Data elements that will template onto ZIP matrix for const S.
42  /// zip_id: <string> # ID of Network Zip. If absent, will create a new zip.
43  /// phases: <phases > # Phases of new ZIP, see above.
44  /// scale_factor_Y: <real > # Scale factor applied to constant Y.
45  /// scale_factor_I: <real > # Scale factor applied to constant I.
46  /// scale_factor_S: <real > # Scale factor applied to constant S.
47  /// ~~~
48  /// @}
