189  template<typename T> T mapPhases(const T& srcVals, const Phases& srcPhases, const Phases& dstPhases)
190  {
191  T result = MapPhasesTraits<T>::init(dstPhases.size());
192  for (std::size_t i = 0; i < srcPhases.size(); ++i)
193  {
194  std::uint8_t j = dstPhases.index(srcPhases[i]);
195  if (j != Phases::noSuchPhase)
196  {
197  result[j] = srcVals[i];
198  }
199  }
200  return result;
201  }

208  template<typename T> arma::Mat<T> mapPhases(const arma::Mat<T>& srcVals,
209  const Phases& srcPhases, const Phases& dstPhases)
210  {
211  arma::Mat<T> result = MapPhasesTraits<arma::Mat<T>>::init(dstPhases.size());
212  for (std::size_t iSrc = 0; iSrc < srcPhases.size(); ++iSrc)
213  {
214  std::uint8_t iDst = dstPhases.index(srcPhases[iSrc]);
215  if (iDst != Phases::noSuchPhase)
216  {
217  for (std::size_t jSrc = 0; jSrc < srcPhases.size(); ++jSrc)
218  {
219  std::uint8_t jDst = dstPhases.index(srcPhases[jSrc]);
220  if (jDst != Phases::noSuchPhase)
221  {
222  result(iDst, jDst) = srcVals(iSrc, jSrc);
223  }
224  }
225  }
226  }
227  return result;
228  }

235  /// @brief Move lower diagonal elements to upper.
236  ///
237  /// This is useful in making sure delta-type loads are correctly represented by an upper triangular matrix,
238  /// as required by the load model.
239  template<typename T> arma::Mat<T>& makeUpper(arma::Mat<T>& m)
240  {
241  for (arma::uword i = 0; i < m.n_rows; ++i)
242  {
243  for (arma::uword j = 0; j < i; ++j)
244  {
245  m(j, i) += m(i, j);
246  m(i, j) = 0;
247  }
248  }
249  return m;
250  }
