168  /// @brief Utility container type combining aspects of map and vector.
169  /// @ingroup Components
170  ///
171  /// May be iterated over or indexed by a string key or an integer index.
172  /// `ComponentPtr` and `ConstComponentPtr` are a kind of smart pointer to elements in the container.
173  /// They are effectively equivalent to storing an iterator to a map, and as such, are stable under
174  /// changes to the mapped type. Thus we can obtain a `ComponentPtr`, and later replace the value
175  /// stored at the given key with another value, and the `ComponentPtr` will reference the newly replaced
176  /// value.
177  template<typename T> class ComponentCollection
178  {
179  public:
180  using ConstPtr = ConstComponentPtr<T>;
181  using Ptr = ComponentPtr<T>;
182 
183  public:
184  size_t size() const {return vec_.size();}
185 
186  ConstPtr operator[](const std::string& key) const
187  {
188  auto it = map_.find(key);
189  if (it != map_.end()) return it; else return ConstPtr();
190  // Ternary causes possible complications.
191  }
192  Ptr operator[](const std::string& key)
193  {
194  auto it = map_.find(key);
195  if (it != map_.end()) return it; else return Ptr();
196  // Ternary causes possible complications.
197  }
198 
199  ConstPtr operator[](size_t idx) const {return vec_[idx];}
200  Ptr operator[](size_t idx) {return vec_[idx];}
201 
202  ConstPtr front() const {return vec_.front();}
203  Ptr front() {return vec_.front();}
204 
205  ConstPtr back() const {return vec_.back();}
206  Ptr back() {return vec_.back();}
207 
208  auto begin() {return vec_.begin();}
209  auto end() {return vec_.end();}
210 
211  auto begin() const {return vec_.cbegin();}
212  auto end() const {return vec_.cend();}
213 
214  auto cbegin() const {return vec_.cbegin();}
215  auto cend() const {return vec_.cend();}
216 
217  auto rbegin() {return vec_.rbegin();}
218  auto rend() {return vec_.rend();}
219 
220  auto rbegin() const {return vec_.rcbegin();}
221  auto rend() const {return vec_.rcend();}
222 
223  auto rcbegin() const {return vec_.rcbegin();}
224  auto rcend() const {return vec_.rcend();}
225 
226  protected:
227  std::map<std::string, std::shared_ptr<T>> map_;
228  std::vector<Ptr> vec_;

231  /// @ingroup Components
232  template<typename T> class MutableComponentCollection : public ComponentCollection<T>
233  {
234  public:
235  using Ptr = typename ComponentCollection<T>::Ptr;
236  using ConstPtr = typename ComponentCollection<T>::ConstPtr;
237 
238  Ptr insert(const std::string& key, std::shared_ptr<T> comp)
239  {
240  auto it = ComponentCollection<T>::map_.find(key);
241  if (it == ComponentCollection<T>::map_.end())
242  {
243  // Insert new element.
244  it = ComponentCollection<T>::map_.insert(std::make_pair(key, comp)).first;
245  ComponentCollection<T>::vec_.push_back(it);
246  }
247  else
248  {
249  // Replace existing element.
250  it->second = comp;
251  }
252  return it;
253  }
254 
255  Ptr reserve(const std::string& key) {insert(key, std::shared_ptr<T>(nullptr));}
256 
257  template<typename U, typename ... Args>
258  std::pair<Ptr, std::shared_ptr<U>> emplace(const std::string& key, Args&&... args)
259  {
260  auto comp = std::make_shared<U>(std::forward<Args>(args)...);
261  return {insert(key, comp), comp};
262  }
263 
264  std::shared_ptr<T> remove(const std::string& key)
265  {
266  std::shared_ptr<T> result;
267  auto mapIt = ComponentCollection<T>::map_.find(key);
268  if (mapIt != ComponentCollection<T>::map_.end())
269  {
270  result = mapIt->second;
271  ComponentCollection<T>::map_.erase(mapIt);
272  auto vecIt = std::find_if(ComponentCollection<T>::vec_.begin(), ComponentCollection<T>::vec_.end(),
273  [&key](Ptr& p){return p.key() == key;});
274  assert(vecIt != ComponentCollection<T>::vec_.end());
275  ComponentCollection<T>::vec_.erase(vecIt);
276  }
277  return result;
278  }
279  };
