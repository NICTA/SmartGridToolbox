28  /// @brief Convert spherical angles and magnitude to a vector.
29  /// @param angs Struct containing spherical angles of the sun.
30  /// @param mag Magnitude of the vector.
31  /// @return The vector.
32  /// x -> north, y -> east, z -> directly up.
33  inline arma::Col<double>::fixed<3> angsAndMagToVec(const SphericalAngles& angs, double mag)
34  {
35  return {{
36  mag * std::cos(angs.azimuth) * std::sin(angs.zenith),
37  mag * std::sin(angs.azimuth) * std::sin(angs.zenith),
38  mag * std::cos(angs.zenith)}};
39  }
40 
41  /// @brief Convert spherical angles and projection to a vector.
42  /// @param angs struct containing spherical angles of the sun.
43  /// @param proj projection of the vector onto a vertical vector.
44  /// @return The vector.
45  /// x -> north, y -> east, z -> directly up.
46  /// If the vector is interpreted as the normal to an area, then the projection is interpreted as being
47  /// onto the horizontal plane.
48  inline arma::Col<double>::fixed<3> angsAndProjToVec(const SphericalAngles& angs, double proj)
49  {
50  return angsAndMagToVec(angs, proj / std::cos(angs.zenith));
51  }
