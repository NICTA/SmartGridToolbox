40  /// @brief Horizontal clear sky diffuse irradiance, given direct irradiance.
41  /// @param directIrradianceMag Magnitude of direct irradiance, W/m^2.
42  /// @return Horizontal diffuse irradiance.
43  /// Just use 10% of direct magnitude, See http://www.pveducation.org/pvcdrom/properties-of-sunlight/air-mass.
44  inline double skyDiffuseIrradiance(double directIrradianceMag)

46  // See http://www.pveducation.org/pvcdrom/properties-of-sunlight/air-mass

68  /// @brief Solar irradiance components (direct vec, diffuse, ground), W/m^2
69  /// @param angs Solar angles of the sun.
70  /// @param altitudeMeters Altitude in meters, default = 0.
71  /// @return Irradiance components (direct vector, diffuse, ground).
72  inline Irradiance irradiance(const SphericalAngles& angs, double altitudeMeters = 0.0)
73  {
74  auto dir = directIrradianceMag(angs, altitudeMeters);
75  auto diff = skyDiffuseIrradiance(dir);
76  auto ground = groundDiffuseIrradiance(dir);
77  return {angsAndMagToVec(angs, dir), diff, ground};
78  }
