171  /// @brief Soonest/smallest rank goes first. If equal rank and time, then sort on the id.
172  class ScheduledUpdatesCompare
173  {
174  public:
175  bool operator()(const std::pair<ComponentPtr<SimComponent>, Time>& lhs,
176  const std::pair<ComponentPtr<SimComponent>, Time>& rhs) const
177  {
178  return ((lhs.second < rhs.second) ||
179  (lhs.second == rhs.second && lhs.first->rank() < rhs.first->rank()) ||
180  (lhs.second == rhs.second && lhs.first->rank() == rhs.first->rank() &&
181  (lhs.first->id() < rhs.first->id())));
182  }
183  };

185  // Smallest rank goes first. If equal rank, then sort on the id.
186  class ContingentUpdatesCompare
187  {
188  public:
189  bool operator()(const ComponentPtr<SimComponent>& lhs, const ComponentPtr<SimComponent>& rhs) const
190  {
191  return ((lhs->rank() < rhs->rank()) ||
192  ((lhs->rank() == rhs->rank()) && (lhs->id() < rhs->id())));
193  }
194  };
