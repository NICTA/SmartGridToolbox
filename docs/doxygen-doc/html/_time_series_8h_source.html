118  /// @brief TimeSeries that changes in a stepwise manner between tabulated times.
119  /// @ingroup SimCore
120  template<typename T, typename V>
121  class StepwiseTimeSeries : public DataTimeSeries<T, V>
122  {
123  protected:
124  using DataTimeSeries<T, V>::points_;
125  using DataTimeSeries<T, V>::defaultV_;
126 
127  public:
128  StepwiseTimeSeries(const V& defaultV) : DataTimeSeries<T, V>(defaultV) {}
129 
130  virtual V value(const T& t) const override
131  {
132  return (points_.size() == 0 || t < points_.begin()->first || t > points_.rbegin()->first)
133  ? defaultV_
134  : (--points_.upper_bound(t))->second; // Highest point <= t.
135  }
136  };

138  /// @brief TimeSeries that uses linear interpolation between tabulated times.
139  /// @ingroup SimCore
140  template<typename T, typename V>
141  class LerpTimeSeries : public DataTimeSeries<T, V>
142  {
143  protected:
144  using DataTimeSeries<T, V>::points_;
145  using DataTimeSeries<T, V>::defaultV_;
146 
147  public:
148  LerpTimeSeries(const V& defaultV) : DataTimeSeries<T, V>(defaultV) {}
149 
150  virtual V value(const T& t) const override
151  {
152  if (points_.size() == 0 || t < points_.begin()->first || t > points_.rbegin()->first) return defaultV_;
153 
154  auto pos2 = points_.upper_bound(t);
155  // pos2 -> first point > s. It can't be begin, but could be end (if t == last point).
156  if (pos2 == points_.end())
157  {
158  return points_.rbegin()->second;
159  }
160  else
161  {
162  // We now know that t is strictly inside the range.
163  auto pos1 = pos2; --pos1;
164  return pos1->second + (pos2->second - pos1->second) * TsTraits<T>::toDouble(t - pos1->first) /
165  TsTraits<T>::toDouble(pos2->first - pos1->first);
166  }
167  }
168  };
